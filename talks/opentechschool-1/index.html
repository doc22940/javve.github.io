List.js

Add search, sort and flexibility to plain HTML lists or tables with cross-browser native JavaScript.

And it's small!

Cross-browser native JavaScript?

  • Works in all modern browsers and down to IE6
  • No dependencies (no jQuery, Prototype, etc)

Example

  • Jonny

  • Beatrice


<div>
    <ul>
       <li>
           <h4>Jonny</h4>
       </li>
       <li>
           <h4>Beatrice</h4>
       </li>
    </ul>
</div>

Add List.js


<div id="persons">
    <ul class="list">
       <li>
           <h4 class="name">Jonny</h4>
       </li>
       <li>
           <h4 class="name">Beatrice</h4>
       </li>
    </ul>
</div>
                        

var options = {
    valueNames: [ 'name' ]
};
var personList = new List('persons', options);

Woah!

Li, div, table, whatever


<div id="persons">
    <table>
        <tbody class="list">
            <tr>
                <td class="name">Jonny</td>
                <td class="city">Stockholm</td>
            </tr>
            <tr>
                <td class="name">Beatrice</td>
                <td class="city">Berlin</td>
            </tr>
        </tbody>
    </table>
</div>
                        

var options = {
    valueNames: [ 'name', 'city' ]
};
var personList = new List('persons', options);

HTML as value


<div id="persons">
    <ul class="list"></ul>
</div>
                    

var options = {
    valueNames: [ 'name', 'image' ],
    item: '<li> \
        <span class="image"></span> \
        <h2 class="name"></h2> \
    </li>';
};
var personList = new List('persons', options);

personList.add({
    name: 'Jonny',
    image: '<img src="http://url.to/cool/image.png">'
});
